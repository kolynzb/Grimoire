## Component

import { useState } from "react";

export const Counter = () => {
  const [count, setCount] = useState(0);
  return (
    <div>
      <button
        onClick={() => setCount(count + 1)}
        className="rounded-md px-2 py-6  border-2  mt-9 border-gray-100"
      >
        Clicked {count} times
      </button>
    </div>
  );
};

<Counter />
